    #include <iostream>
using namespace std;
int res = 0;
int all[5010];
void cal(int start, int end) {
    if (start >= end) return;
    int i, min = 999999999, last = start; \\这里的last当时赋值为0了
    for (i = start; i < end; i++) {
        if (all[i] > end - start) {
            res += 1;
            cal(start, i);
            cal(i+1, end);
            return;
        }
        if (all[i] < min) min = all[i];
    }
    res += min;
    for (i = start; i < end; i++) {
        all[i] = all[i] - min;
        if (all[i] == 0) {
            cal(last, i);
            last = i + 1;
        }
        if (i == end - 1) cal(last, end);
    }
}
int main() {
    int i, n;
    scanf("%d", &n);
    for (i = 0; i < n; i++) scanf("%d", &all[i]);
    cal(0, n);
    printf("%d", res);
    return 0;
}
